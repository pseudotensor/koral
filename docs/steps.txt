################
# Make and Run Koral
#

# Setting OPENMP:
export OMP_NUMTHREADS=8

# Editing:
# problem.h - to choose problem, e.g., #define PROBLEM 6 for RADTUBE)
# PROBLEMS/RADTUBE/define.h - problem parameters
# PROBLEMS/RADTUBE/init.h - initial state
# PROBLEMS/RADTUBE/bc.c - boundary conditions
# PROBLEMS/RADTUBE/kappa.c, kappaes.c - opacities

# Compiling:
# make clean required when any of the files inside PROBLEMS has changed
make clean ; make
# or ./mk

# Running
mkdir run ; rm -rf run/* ; cd run ; cp -a ../ko ../PROBLEMS . ; mkdir gifs dumps ; ./ko
